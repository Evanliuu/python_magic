<<显示所有任务进程>>
    tasklist

<<关闭指定的任务进程>>
    taskkill /im python.exe /f
    taskkill /pid 11111

<<使用vbs调用bat文件可以隐藏cmd框>>
    Set obj = CreateObject("Wscript.Shell")
    obj.run "cmd /c C:\Users\evan\Desktop\test.bat", vbhide

<<使用vbs关闭指定的任务进程>>
    msgbox "确认关闭？"
    Set obj = CreateObject("Wscript.Shell")
    obj.run "taskkill /im wscript.exe /f", 0

<<使用vbs循环模拟点击键盘>>
    msgbox "确认打开？"
    Set obj = CreateObject("Wscript.Shell")
    do while true
        Wscript.Sleep 1000*120   # 1000等于1秒，此注释在使用时要删除
        obj.SendKeys "^{ESC}"
        Wscript.Sleep 50
        obj.SendKeys "^{ESC}"
        loop
